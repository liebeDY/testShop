package com.example.demo.member;

import java.sql.Date;

public class Member {

	private int num;
	private String id;
	private String pwd;
	private String name;
	private String nickname;
	private String tel;
	private String email;
	private String postcode;
	private Date reg_date;
	private String addr1;
	private String addr2;
	private int type;
	
	public Member() {
		super();
	}

	public Member(int num, String id, String pwd, String name, String nickname, String tel, String email,
			String postcode, Date reg_date, String addr1, String addr2, int type) {
		super();
		this.num = num;
		this.id = id;
		this.pwd = pwd;
		this.name = name;
		this.nickname = nickname;
		this.tel = tel;
		this.email = email;
		this.postcode = postcode;
		this.reg_date = reg_date;
		this.addr1 = addr1;
		this.addr2 = addr2;
		this.type = type;
	}

	public int getNum() {
		return num;
	}

	public void setNum(int num) {
		this.num = num;
	}

	public String getId() {
		return id;
	}

	public void setId(String id) {
		this.id = id;
	}

	public String getPwd() {
		return pwd;
	}

	public void setPwd(String pwd) {
		this.pwd = pwd;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public String getNickname() {
		return nickname;
	}

	public void setNickname(String nickname) {
		this.nickname = nickname;
	}

	public String getTel() {
		return tel;
	}

	public void setTel(String tel) {
		this.tel = tel;
	}

	public String getEmail() {
		return email;
	}

	public void setEmail(String email) {
		this.email = email;
	}

	public String getPostcode() {
		return postcode;
	}

	public void setPostcode(String postcode) {
		this.postcode = postcode;
	}

	public Date getReg_date() {
		return reg_date;
	}

	public void setReg_date(Date reg_date) {
		this.reg_date = reg_date;
	}

	public String getAddr1() {
		return addr1;
	}

	public void setAddr1(String addr1) {
		this.addr1 = addr1;
	}

	public String getAddr2() {
		return addr2;
	}

	public void setAddr2(String addr2) {
		this.addr2 = addr2;
	}

	public int getType() {
		return type;
	}

	public void setType(int type) {
		this.type = type;
	}

	@Override
	public String toString() {
		return "Member [num=" + num + ", id=" + id + ", pwd=" + pwd + ", name=" + name + ", nickname=" + nickname
				+ ", tel=" + tel + ", email=" + email + ", postcode=" + postcode + ", reg_date=" + reg_date + ", addr1="
				+ addr1 + ", addr2=" + addr2 + ", type=" + type + "]";
	}

	



	
	
	
}
